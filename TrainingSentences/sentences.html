function getParam(name){ const url = new URL(window.location.href); return url.searchParams.get(name) || ""; } const topic = getParam("topic"); const topicTitle = document.getElementById("topicTitle"); const rows = document.getElementById("rows"); const qInput = document.getElementById("q"); const levelSel = document.getElementById("level"); const regSel = document.getElementById("register"); const metaInfo = document.getElementById("metaInfo"); let SENTENCES = []; // خريطة أسماء المستوى (سطر واحد) const LEVEL_LABEL = { "A1": "A1 — مبتدئ", "A2": "A2 — أساسي", "B1": "B1 — متوسط", "B2": "B2 — فوق المتوسط" }; // خريطة الأسلوب للعربية const REG_AR = { "formal": "رسمي", "street_light": "Street Light", "neutral": "محايد" }; async function load(){ if(!topic){ topicTitle.textContent = "— لا يوجد موضوع محدد"; return; } // العنوان من الميتاداتا try { const metaRes = await fetch("data/topics_meta.json"); const meta = await metaRes.json(); const t = meta.find(m => m.key === topic); topicTitle.textContent = t ? t.title : topic; } catch(e){ topicTitle.textContent = topic; } // قراءة الجمل من مجلد البيانات الموحد try { const res = await fetch(`data/sentences/${encodeURIComponent(topic)}.json`); SENTENCES = await res.json(); } catch(e){ console.error("Failed to load sentences:", e); SENTENCES = []; } render(); } function render(){ const q = (qInput.value || "").trim().toLowerCase(); const lvl = levelSel.value || ""; const reg = regSel.value || ""; const filtered = SENTENCES.filter(s => { let ok = true; if (q) { const hay = `${s.en} ${s.ar} ${s.ar_pron}`.toLowerCase(); ok = hay.includes(q); } if (ok && lvl) ok = (s.level === lvl); if (ok && reg) ok = (s.register === reg); return ok; }); rows.innerHTML = ""; for (const s of filtered){ const levelText = LEVEL_LABEL[s.level] || s.level || ""; const regText = REG_AR[s.register] || s.register || ""; const tr = document.createElement("tr"); tr.innerHTML = ` ${escapeHTML(levelText)} ${escapeHTML(regText)} ${escapeHTML(s.en || "")} ${escapeHTML(s.ar || "")} ${escapeHTML(s.ar_pron || "")} ${(s.tags||[]).map(t=>`${escapeHTML(String(t))}`).join(" ")} `; rows.appendChild(tr); } metaInfo.textContent = `عدد الجمل المعروضة: ${filtered.length} / الإجمالي: ${SENTENCES.length}`; } function escapeHTML(str){ return String(str).replace(/[&<>"']/g, m => ({ "&":"&","<":"<",">":">","\"":""","'":"'" }[m])); } qInput.addEventListener("input", render); levelSel.addEventListener("change", render); regSel.addEventListener("change", render); load();
